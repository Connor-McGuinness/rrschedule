RRSchedule

A gem for ruby that makes it easier to generate Round-Robin schedules

      gem install rrschedule
      
      schedule = RRSchedule::Schedule.new(
        :teams => ["Jazz","Rocks","Chargers","Sprinklers","Pacers","Magicians","Ferrets","Munchkins"],
        :playing_surfaces => ["Big court", "Small court"],
        :game_times => ["5:00 PM","7:00 PM"],
        :cycles => 1,
        :shuffle_initial_order => true,
        :start_date => Date.today, 
        :wdays => [3] #games are played on wednesdays...
      )
      
      schedule.generate
      
      #human readable schedule
      puts schedule.to_s
      
      #Or, if you want to loop through the schedule...
      schedule.gamedays.each do |gd,games|
        #...        
        games.each do |game|
          #...
        end
      end
See github page for more info.
